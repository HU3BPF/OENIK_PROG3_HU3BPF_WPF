111  join product in this.iRepositoryProduct.GetALL()
112  on brand.BrandId equals product.BrandrId
113  let item = new { BrandName = brand.BrandName, BrandId = brand.BrandId, Price = product.ProductPrice }

115  orderby grp.Key descending
116  select new BrandAveragerProductPrice
